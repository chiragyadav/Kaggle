def plot_indicator(indicator_code,ylabel=None,title=None):
    indicator_name= str(df[df.IndicatorCode==indicator_code]['IndicatorName'].unique()[0])
    if(title==None):
        title=indicator_name.split('(')[0]
    if(ylabel==None):
        if(len(indicator_name.split('('))>1):
            ylabel= indicator_name.split('(')[1].replace(')','')
        else:
            ylabel = indicator_name
        
    df_indicator = df[(df.CountryName=='India')&(df.IndicatorCode==indicator_code)]
    plt.plot(df_indicator.Year,df_indicator.Value,'o-',color='g')
    plt.xlabel('Years')
    plt.ylabel(ylabel)
    plt.title(title)
    plt.show()
